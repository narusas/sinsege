function getThumbNailImgList(data) {

var dataList = data.result;

$("end").value = dataList.length;


util = new PageUtil();

util.totalCnt= data.tot_cnt; 
util.pageRows= eval($("max").value);     
util.curPage= eval($("skip").value);  
util.disPagepCnt= 1;     
util.setTotalPage();

var img_text = "";

if(util.getPrev() != util.curPage) {
img_text += "<li class='fl_le_first'><a href='javascript:go_page(" + util.getPrev() + ");'><img src='/ret/assets/images/common/btn_photo_prev.gif' height='58' /></a></li>";
} else {
img_text += "<li class='fl_le_first'><img src='/ret/assets/images/common/btn_photo_prev.gif' height='58' /></li>";
}

var seq_no;

for(var i=0 ; i<util.pageRows ; i++) {

num = $("skip").value + "" + i;

if(dataList[i] != null) {
var rs = dataList[i];
img_text += "<li class='fl_le'><a href='javascript:selectThumb(" + num + ", "+rs.SEQ_NO+");'><img id='thumb" + num + "' src='" + rs.IMG_DATA + "' width='82' height='53' style='border: solid 1px #666666'/></a></li>";
if(i == 0) seq_no = rs.SEQ_NO;
} else {
img_text += "<li class='fl_le'><img id='thumb" + num + "' src='/ret/assets/images/common/blank.png' width='88' height='53'/></li>";
}

}


if(util.curPage < util.getNext()) {
img_text += "<li class='fl_le'><a href='javascript:go_page(" + util.getNext() + ");'><img src='/ret/assets/images/common/btn_photo_next.gif' height='58' /></a></li>";
} else {
img_text += "<li class='fl_le'><img src='/ret/assets/images/common/btn_photo_next.gif' height='58' /></li>";
}


$("thumb").innerHTML = img_text;

if(dataList.length > 0)
selectThumb($("skip").value + "" + 0, seq_no);
else {
$("img_data").innerHTML = "<img src='/ret/assets/images/common/blank.png' width='375' height='240' />";
}
}
